	\hat{H} = - J \sum_{\langle i,j\rangle}
		\hat{\sigma}^x_i \hat{\sigma}^x_j
	- h \sum_j \hat{\sigma}_j^z
\end{equation*}
where $J,h>0$ with PBC $\hat{\sigma}_L^x \hat{\sigma}_{L+1}^x = \hat{\sigma}_{L}^x \hat{\sigma}_1^x$.
