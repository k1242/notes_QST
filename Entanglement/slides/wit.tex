\onslide<2,3>{

\phantom{42}

A state $\rho_{AB}$ is entangled if and only if a positive map $\Lambda$ exists: 
\vspace{-2mm}
\begin{equation*}
	\left(\mathbbm{1}_A \otimes \Lambda_B \right) \rho_{AB} < 0.
\end{equation*}

}


\onslide<3>{

% \phantom{42}

Peres-Horodecki criterion of being separable:
\vspace{-2mm}
\begin{equation*}
	\rho_{AB}^{T_B} \geq 0.
\end{equation*}

}
